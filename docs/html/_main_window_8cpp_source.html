471void MainWindow::setupButtonAnimation(QPushButton* button)
472{
480 class AnimatedButton : public QObject
481 {
482 public:
483 QPushButton* button;
484 QPropertyAnimation* shakeAnimation;
485 QRect originalGeometry;
486 bool isHovering;
487
488 AnimatedButton(QPushButton* btn) : QObject(btn), button(btn), isHovering(false)
489 {
490 originalGeometry = button->geometry();
491
492 // Create smooth shake animation using geometry property
493 shakeAnimation = new QPropertyAnimation(button, "geometry", this);
494 shakeAnimation->setDuration(800);
495 shakeAnimation->setEasingCurve(QEasingCurve::InOutSine);
496 shakeAnimation->setLoopCount(-1); // Loop infinitely while hovering
497 }
498
499 protected:
500 bool eventFilter(QObject* obj, QEvent* event) override
501 {
502 if (obj == button) {
503 if (event->type() == QEvent::Enter) {
504 startHoverAnimation();
505 } else if (event->type() == QEvent::Leave) {
506 stopHoverAnimation();
507 }
508 }
509 return QObject::eventFilter(obj, event);
510 }
511
512 private:
513 void startHoverAnimation()
514 {
515 isHovering = true;
516 originalGeometry = button->geometry();
517
518 // Scale up the button using stylesheet
519 QString currentStyle = button->styleSheet();
520 QString newStyle = currentStyle + " QPushButton { transform: scale(1.1); }";
521 button->setStyleSheet(newStyle);
522
523 // Resize the button geometry for the scale effect
524 QRect scaledGeometry = originalGeometry.adjusted(-10, -5, 10, 5);
525 button->setGeometry(scaledGeometry);
526
527 // Create smooth shake with more keyframes for smoother motion
528 QRect shakeLeft = scaledGeometry.adjusted(-3, 0, -3, 0);
529 QRect shakeRight = scaledGeometry.adjusted(3, 0, 3, 0);
530
531 shakeAnimation->setStartValue(scaledGeometry);
532 shakeAnimation->setKeyValueAt(0.125, shakeLeft);
533 shakeAnimation->setKeyValueAt(0.25, scaledGeometry);
534 shakeAnimation->setKeyValueAt(0.375, shakeRight);
535 shakeAnimation->setKeyValueAt(0.5, scaledGeometry);
536 shakeAnimation->setKeyValueAt(0.625, shakeLeft);
537 shakeAnimation->setKeyValueAt(0.75, scaledGeometry);
538 shakeAnimation->setKeyValueAt(0.875, shakeRight);
539 shakeAnimation->setEndValue(scaledGeometry);
540
541 shakeAnimation->start();
542 }
543
544 void stopHoverAnimation()
545 {
546 isHovering = false;
547 shakeAnimation->stop();
548
549 // Reset stylesheet to remove scale
550 QString currentStyle = button->styleSheet();
551 currentStyle.remove(" QPushButton { transform: scale(1.1); }");
552 button->setStyleSheet(currentStyle);
553
554 // Return to original size smoothly
555 QPropertyAnimation* returnAnimation = new QPropertyAnimation(button, "geometry", this);
556 returnAnimation->setDuration(150);
557 returnAnimation->setEasingCurve(QEasingCurve::OutCubic);
558 returnAnimation->setStartValue(button->geometry());
559 returnAnimation->setEndValue(originalGeometry);
560
561 connect(returnAnimation, &QPropertyAnimation::finished, returnAnimation, &QPropertyAnimation::deleteLater);
562 returnAnimation->start();
563 }
564 };
565
566 AnimatedButton* animatedButton = new AnimatedButton(button);
567 button->installEventFilter(animatedButton);
568}
